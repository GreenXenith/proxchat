#    Distance at which chat from nearby players is "heard".

#    Show the player distance in chat messages.
proxchat.show_distance (Player distance in chat) bool false

#    Privileges required to use /global (comma-separated list, empty for anyone).
proxchat.global_privs (/global required privs) string shout

#    Whether or not players may configure their own hear distance.
proxchat.player_config (Player distance configuration) bool false
